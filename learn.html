// DOM Elements const cityInput = document.getElementById("cityInput"); const searchBtn = document.getElementById("searchBtn"); const weatherResult = document.getElementById("weatherResult"); const apiKey = "d64ab0aac5d6fddede6262ade0decd06"; const baseUrl = "https://api.openweathermap.org/data/2.5/weather"; // ARRAY to store searched cities const searchedCities = []; async function fetchWeather(city) { const url = `${baseUrl}?q=${city}&appid=${apiKey}&units=metric`; const response = await fetch(url); const data = await response.json(); return data; } function createWeatherCard(data) { if (data.cod !== 200) { alert("City not found!"); return; } const card = document.createElement("div"); card.className = "weather-card"; // API icon const iconCode = data.weather[0].icon; const iconUrl = `https://openweathermap.org/img/wn/${iconCode}@2x.png`; // console.log(data.weather[0].icon); card.innerHTML = `
${data.name} ${data.sys.country}
${Math.round(data.main.temp)}°C
${data.weather[0].description.toUpperCase()}
`; // DELETE BUTTON const deleteBtn = card.querySelector(".delete-btn"); deleteBtn.addEventListener("click", () => { const cityName = data.name.toLowerCase(); const index = searchedCities.indexOf(cityName); if (index > -1) { searchedCities.splice(index, 1); } card.remove(); }); weatherResult.appendChild(card); } searchBtn.addEventListener("click", async () => { const city = cityInput.value.trim().toLowerCase(); if (!city) return; if (searchedCities.includes(city)) { alert("City already added"); return; } const data = await fetchWeather(city); createWeatherCard(data); // STORE CITY IN ARRAY searchedCities.push(city); cityInput.value = ""; }); cityInput.addEventListener("keydown", (e) => { if (e.key === "Enter") { searchBtn.click(); } });
